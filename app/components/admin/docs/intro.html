Angular Global $rootScope variables and functions.

All $scope objects have the following variables and functions. 

             //The account of the currently logged in user
             $scope.account = {
                login:'billyb',
                firstname:'Billy',
                lastName:'Bob',
                email:'billyBob@gmail.com',
                roles:['ROLE_ADMIN','ROLE_SUPER_USER','ROLE_USER'],
             }
             //The currently recognized roles
             $scope.userRoles = {
                    all: '*',
                    admin: 'ROLE_ADMIN',
                    superUser:'ROLE_SUPER_USER',
                    user: 'ROLE_USER'
             });
             //Returns true if the current logged in user has admin privileges
             $scope.isAdmin();
             //Returns true if the current logged in user has super user privileges
             $scope.isSuperUser();
             //Returns true if the current logged in user has the role provided
             $scope.isAuthorized(role);
             //Returns true if the user is currently logged in
             $scope.authenticated;
Example usage in an HTML file 

             <li ng-show="isAdmin()">
                <a ui-sref="admin">{{account.firstName}}</a>
             </li>
